isPrime

    isPrime = 1
    currentFactor = currentNumber - 1
    factorsLeft = currentNumber - 2
    while(factorsLeft){
        if(! currentNumber % currentFactor){
            isPrime = 0
        }
        factorsLeft--;
        currentFactor--;
    }
